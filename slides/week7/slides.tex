\hypertarget{slide-1-introduction-to-eigenvalues}{%
\subsection{Slide 1: Introduction to
Eigenvalues}\label{slide-1-introduction-to-eigenvalues}}

\begin{itemize}
\tightlist
\item
  Eigenvector \(\mathbf{v}\):

  \begin{itemize}
  \tightlist
  \item
    A vector that only changes by a scalar factor \(\lambda\).
    \[A\mathbf{v} = \lambda\mathbf{v}\].
  \end{itemize}
\item
  Reference Video:
  \href{https://www.youtube.com/watch?v=PFDu9oVAE-g}{MIT OCW Video}.
\end{itemize}

\hypertarget{slide-2-computing-eigenvalues}{%
\subsection{Slide 2: Computing
Eigenvalues}\label{slide-2-computing-eigenvalues}}

\begin{itemize}
\tightlist
\item
  Solve \[\det(A - \lambda I) = 0\].
\item
  Example with \[
  \]A =

  \begin{bmatrix} 4 & 2 \\ $$
  1 & 3 \end{bmatrix}

  \$\$.
\end{itemize}
